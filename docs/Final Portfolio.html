import pandas as pd
import altair as alt

# Load the dataset from the URL
data_url = 'https://calvin-data304.netlify.app/data/swd-lets-practice-ex-5-03.csv'
data = pd.read_csv(data_url)

# Convert the 'Date' column to datetime and sort the DataFrame by date
data['Date'] = pd.to_datetime(data['Date'])
data = data.sort_values('Date')

# Calculate the total touchpoints for each month
data['Total Touchpoints'] = data[['Phone Touchpoints', 'Chat Touchpoints', 'Email Touchpoints']].sum(axis=1)

# Calculate the growth percentage using the first month as the baseline
baseline_value = data['Total Touchpoints'].iloc[0]
data['Growth Percentage'] = ((data['Total Touchpoints'] - baseline_value) / baseline_value) * 100

# Create the line chart
growth_chart = alt.Chart(data).mark_line(point=True).encode(
    x=alt.X('Date:T', axis=alt.Axis(title='Month')),
    y=alt.Y('Growth Percentage:Q', axis=alt.Axis(title='Growth Percentage Relative to Jan 2018 (%)')),
    tooltip=['Date:T', 'Growth Percentage:Q']
).properties(
    title='Monthly Growth Percentage of Total Customer Touchpoints Since Jan 2018'
)

# Display the chart in your local Jupyter notebook or save it as an HTML file
growth_chart.display()

This is the percentage growth rate of the total number of Touchpoints over time based on January 2018. This chart can more clearly see the specific numerical changes over time, as well as the growth and decline trends of the data in different months.

import pandas as pd
import altair as alt

data_url = 'https://john-ju199.github.io/DATA-304/tanzania_dhs_data.csv'
data = pd.read_csv(data_url)

# Function to calculate the datetime representation of the survey's midpoint year
def calculate_datetime_midpoint(year_string):
    if '-' in year_string:
        start_year, end_year = year_string.split('-')
        midpoint_year = int(start_year) + (int(end_year) - int(start_year)) / 2
    else:
        midpoint_year = int(year_string)
    # Return the January 1st of the closest integer year to the midpoint
    return pd.to_datetime(f'{int(midpoint_year)}-01-01')

# Apply the function to calculate the datetime midpoint
data['Survey_Date'] = data['DHS Survey Year'].apply(calculate_datetime_midpoint)

# Define base chart with common encodings
base = alt.Chart(data).encode(
    alt.X('Survey_Date:T', title='Survey Year')  # Use the datetime field for the x-axis
)

# Define individual charts for each metric
fertility_chart = base.mark_line(point=True, color='blue').encode(
    alt.Y('Total Fertility Rate, All Women Ages 15-49:Q', title='Total Fertility Rate (15-49)'),
    tooltip=['DHS Survey Year', 'Total Fertility Rate, All Women Ages 15-49']
).properties(
    title='Total Fertility Rate over Survey Years'
)

contraception_chart = base.mark_line(point=True, color='green').encode(
    alt.Y('Current Use of a Modern Method of Contraception-All Women (%):Q', title='Contraception Use (%)'),
    tooltip=['DHS Survey Year', 'Current Use of a Modern Method of Contraception-All Women (%)']
).properties(
    title='Contraception Use over Survey Years'
)

unmet_need_chart = base.mark_line(point=True, color='red').encode(
    alt.Y('Unmet Need for Family Planning-All Women (%):Q', title='Unmet Need for Family Planning (%)'),
    tooltip=['DHS Survey Year', 'Unmet Need for Family Planning-All Women (%)']
).properties(
    title='Unmet Need for Family Planning over Survey Years'
)

# Combine the charts vertically with shared x-axis scales
combined_chart = alt.vconcat(fertility_chart, contraception_chart, unmet_need_chart).resolve_scale(
    x='shared'
).properties(
    title="Trends in Fertility Rate, Contraception Use, and Unmet Family Planning Needs in Tanzania"
)

combined_chart

The three vertically concatenated charts effectively highlight Tanzania’s changing fertility rates, contraceptive usage, and family planning needs from 1991 to 2016. With each chart focusing on a specific aspect—showing declining fertility, increased contraceptive use, and variable unmet needs—this visualization provides valuable insights for policymakers, health workers, and leaders to address Tanzanian women’s reproductive health and family planning concerns.

import pandas as pd
import altair as alt

data_url = 'https://www.ncei.noaa.gov/data/global-marine/doc/Marine_CSV_sample.csv'
data = pd.read_csv(data_url)

# Convert 'Time of Observation' to datetime
data['Time of Observation'] = pd.to_datetime(data['Time of Observation'])

# Function to categorize time of day
def categorize_time_of_day(hour):
    if 6 <= hour < 12:
        return 'Morning'
    elif 12 <= hour < 18:
        return 'Afternoon'
    elif 18 <= hour < 24:
        return 'Evening'
    else:
        return 'Night'

# Apply the function to the 'Time of Observation' column to create 'Time of Day'
data['Time of Day'] = data['Time of Observation'].dt.hour.apply(categorize_time_of_day)

# Create a slider for selecting wind speed range
wind_speed_slider = alt.binding_range(min=data['Wind Speed'].min(), max=data['Wind Speed'].max(), step=1, name='Wind Speed: ')
slider_selection = alt.selection_single(bind=wind_speed_slider, fields=['Wind Speed'], name="Selector")

# Enhance the chart by varying point sizes based on 'Sea Level Pressure' and adjusting other aesthetic elements
# and add the slider for interactive wind speed filtering
chart = alt.Chart(data).mark_point().encode(
    x='Longitude:Q',
    y='Latitude:Q',
    color='Wind Speed:Q',
    size='Sea Level Pressure:Q',
    tooltip=['Identification', 'Time of Observation', 'Wind Speed', 'Sea Level Pressure']
).add_selection(
    slider_selection
).transform_filter(
    slider_selection
).properties(
    width=250,
    height=250
).facet(
    facet='Time of Day:N',
    columns=2
).interactive()

# Set the title
chart = chart.properties(
    title="Marine Observations Throughout the Day: Dynamics of Wind Speed and Sea Level Pressure"
)
